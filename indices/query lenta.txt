SELECT EMP.NMEMPFANT ,
       SUM(EXT.QTMVTDSP) + SUM(EXT.QTMVTBLQ) + SUM(EXT.QTMVTRSV) AS QUANTFIS ,
       SUM(EXT.QTMVTBLQ) AS QUANTBLQ ,
       SUM(EXT.QTMVTRSV) AS QUANTRES ,
       SUM(EXT.QTMVTDSP) AS QUANTEST
FROM dbo.TBMVT001EXT EXT WITH(NOLOCK)
INNER JOIN TBEMP001CAD EMP WITH(NOLOCK) ON EXT.IDEMPCAD = EMP.IDEMPCAD
INNER JOIN TBPRD001CAD PRD WITH(NOLOCK) ON EXT.IDPRDCAD = PRD.IDPRDCAD
INNER JOIN TBPRD001CFG CFG WITH(NOLOCK) ON CFG.IDPRDCAD = PRD.IDPRDCAD
AND CFG.IDEMPCAD = EMP.IDEMPCAD
AND CFG.STPRDCFGEMP = 5
INNER JOIN TBEMP001PRM PRM WITH(NOLOCK) ON EMP.IDEMPCAD = PRM.IDEMPCAD
LEFT JOIN TBPRD001PNC PNC WITH(NOLOCK) ON PRM.IDLSTPNC = PNC.IDLSTPNC
LEFT JOIN TBPRD001PNI PNI WITH(NOLOCK) ON PNC.IDLSTPNC = PNI.IDLSTPNC
AND PNI.STLSTPNI = 5
AND PNI.IDPRDCAD = EXT.IDPRDCAD
WHERE EXT.DTMVTEXT IN
    (SELECT MAX(DTMVTEXT)
     FROM TBMVT001EXT EXT1 WITH(NOLOCK)
     WHERE EXT1.IDEMPCAD = EXT.IDEMPCAD
       AND EXT1.IDRMACAD = EXT.IDRMACAD
       AND EXT1.IDPRDCAD = EXT.IDPRDCAD
       AND EXT1.CDEXTNV1 = EXT.CDEXTNV1
       AND EXT1.CDEXTNV2 = EXT.CDEXTNV2
     GROUP BY IDPRDCAD)
  AND PRD.IDPRDCAD = 61691
  AND PRD.STPRDCAD = 5
GROUP BY EMP.CDAUXEMPCAD,
         EMP.NMEMPFANT,
         EMP.IDEMPCAD,
         PRD.IDPRDCAD,
         PNI.VLPREVELST,
         CFG.QTETQMIN,
         CFG.QTETQMAX
